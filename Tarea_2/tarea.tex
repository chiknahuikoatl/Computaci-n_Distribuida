\documentclass{article}

\usepackage{pgf}
\usepackage{tikz}
\usetikzlibrary{arrows,automata}
\usepackage[latin1]{inputenc}
\begin{document}
\newcommand{\ed}[2]{(#1) edge (#2)}
\begin{tikzpicture}[-,>=stealth',auto,
node distance = 2.8cm,
    shorten > = 1pt, semithick,
 state/.style = {circle, fill=#1, draw=black!,
                 inner sep=1mm, text=black},
 state/.default = white]
                        
  \tikzstyle{every state}=[fill=white,draw=black!,text=black]  

  \node (A) [state]                    {$(A,0)$};
  \node (B) [state=gray, right of=A]   {$(B,0)$};
  \node (C) [state, below of=B]        {$(A,1)$};
  \node (D) [state=gray, below of=A]   {$(B,1)$};
  \node (E) [state, above right of=B]  {$(A,0)$};
  \node (F) [state=gray, right of=E]   {$(B,0)$};
  \node (G) [state=gray, below of=E]   {$(B,\frac{1}{3})$};
  \node (H) [state, below of=G]        {$(A,\frac{2}{3})$};
  \node (I) [state=gray, below of=H]   {$(B,1)$};
  \node (J) [state, below of=F]        {$(A,\frac{1}{3})$};
  \node (K) [state=gray, below of=J]   {$(B,\frac{2}{3})$};
  \node (L) [state, below of=K]        {$(A,1)$};

  \path (A) edge               (B)
        (B) edge               (C)
        (C) edge               (D)
        (D) edge               (A)
        \ed{E}{F}
        \ed{E}{G}
        \ed{G}{H}
        \ed{H}{I}
        \ed{F}{J}
        \ed{J}{K}
        \ed{K}{L}
        \ed{L}{I};
        
\end{tikzpicture}

\end{document}
